def multiply(num1: str, num2: str) -> str:
    if num1 == "0" or num2 == "0":
        return "0"
    m, n = len(num1), len(num2)
    res = [0] * (m + n)
    for i in range(m - 1, -1, -1):
        a = ord(num1[i]) - ord('0')
        for j in range(n - 1, -1, -1):
            b = ord(num2[j]) - ord('0')
            prod = a * b
            sum_ = prod + res[i + j + 1]
            res[i + j + 1] = sum_ % 10
            res[i + j] += sum_ // 10
    # skip leading zeros
    idx = 0
    while idx < len(res) and res[idx] == 0:
        idx += 1
    return ''.join(str(d) for d in res[idx:]) if idx < len(res) else "0"
